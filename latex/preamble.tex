% TikZ helper
\newcommand{\dotrow}[3]{% end, label, labelpos
  \begin{scope}[start chain=going right]
    \foreach \i in {0,...,#1}
      \node[on chain, dotnode, label=#3:{\i}] (#2\i) {};
  \end{scope}
}
